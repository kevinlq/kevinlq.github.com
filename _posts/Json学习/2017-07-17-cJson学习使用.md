### cJson结构体

```c
typedef struct cJSON {
       struct cJSON*next,*prev;    /* 遍历数组或对象链的前向或后向链表指针*/
       struct cJSON *child;        /*数组或对象的孩子节点*/
       int type;                   /* key的类型*/
       char *valuestring;          /*字符串值*/
       int valueint;               /* 整数值*/
       double valuedouble;         /* 浮点数值*/
       char *string;               /* key的名字*/
} cJSON;
```
